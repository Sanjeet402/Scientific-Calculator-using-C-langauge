/**  program to store N members sales man details into a RandomAccessFile ransales.dat. 
This file contains name and sales fields.  */
import java.io.RandomAccessFile;
import java.io.FileNotFoundException;
import java.io.IOException;
import java.io.EOFException;
import java.util.Scanner;
class Prog221
{ public static void main(String[] args) throws FileNotFoundException,IOException
{RandomAccessFile raf=new RandomAccessFile("ransales.dat","r");
Scanner sc=new Scanner(System.in);
String n,v,d,sn; int p,c=1;
System.out.print("Enter Search Name : "); sn=sc.nextLine();
try
{ while(true)
{ n=raf.readUTF();v=raf.readUTF();d=raf.readUTF();p=raf.readInt();
if(sn.equalsIgnoreCase(n))
{System.out.println("Name : "+n);
System.out.println("Village : "+v);
System.out.println("District : "+d);
System.out.println("Pincode : "+p);
}
c=c+1;
 }
}
  catch(EOFException e) { raf.close(); }
 }
}
