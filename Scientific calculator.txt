PayrollManual.jsx
import React, { useState } from 'react';
import Navbar from './Navbar';
import DatePicker from "react-datepicker";

import "react-datepicker/dist/react-datepicker.css";

function PayrollManual() {
  // --- State ---
  const [filters, setFilters] = useState({
    debitAccount: "1",
    date: new Date(),
    payrollType: "Domestic",
    paymentCurrency: "INR",
  });

  const [employees, setEmployees] = useState([]);
  const [newEmployee, setNewEmployee] = useState({ id: '', name: '', salary: '', currency: 'INR', accountNumber: '' });
  const [error, setError] = useState('');

  // --- Handlers ---
  const handleFiltersChange = (updated) => setFilters({ ...filters, ...updated });

  const handleAddEmployee = (e) => {
    e.preventDefault();
    if (!newEmployee.id || !newEmployee.name || !newEmployee.salary || !newEmployee.accountNumber) {
      setError('All fields are required.');
      return;
    }
    if (employees.some(emp => emp.id === newEmployee.id)) {
      setError('An employee with this ID already exists.');
      return;
    }
    setEmployees([...employees, newEmployee]);
    setNewEmployee({ id: '', name: '', salary: '', currency: 'INR', accountNumber: '' });
    setError('');
  };

  const handleInputChange = (e) => {
    const { name, value } = e.target;
    if (name === 'id' || name === 'salary' || name === 'accountNumber') {
      if (/^\d*$/.test(value)) setNewEmployee(prev => ({ ...prev, [name]: value }));
    } else if (name === 'name') {
      if (/^[a-zA-Z\s]*$/.test(value)) setNewEmployee(prev => ({ ...prev, [name]: value }));
    } else {
      setNewEmployee(prev => ({ ...prev, [name]: value }));
    }
  };

  const handleRemoveEmployee = (idx) => setEmployees(employees.filter((_, i) => i !== idx));
  const totalAmount = employees.reduce((acc, emp) => acc + Number(emp.salary || 0), 0);

  // --- JSX ---
  return (
    <>
      <Navbar />

      <div className="container my-4">
        <div className="p-4 bg-white shadow-sm rounded-3">
          <header className="pb-3 mb-4 border-bottom">
            <h2 className="text-primary">Payroll Initiation</h2>
            <p className="text-muted">Create payment batches by entering employee details manually.</p>
          </header>

          {/* Filters */}
          {/* ✅ CORRECTED THIS LINE */}
          <div className="row g-3 my-2 p-3 border rounded-3 bg-light align-items-end">
            <div className="col-md-6 col-lg-3">
              <label className="form-label fw-bold">Debit Account</label>
              <select className="form-select" value={filters.debitAccount} onChange={e => handleFiltersChange({ debitAccount: e.target.value })}>
                <option value="">Select Account</option>
                <option value="1">Account 1 - Savings</option>
                <option value="2">Account 2 - Corporate</option>
              </select>
            </div>
            <div className="col-md-6 col-lg-3">
              <label className="form-label fw-bold">Date</label>
              <DatePicker
                selected={filters.date}
                onChange={(date) => handleFiltersChange({ date: date })}
                className="form-control"
                dateFormat="yyyy-MM-dd"
              />
            </div>
            <div className="col-md-6 col-lg-3">
              <label className="form-label fw-bold">Payroll Type</label>
              <div className="btn-group w-100">
                <button className={`btn ${filters.payrollType === "Domestic" ? "btn-primary" : "btn-outline-secondary"}`} onClick={() => handleFiltersChange({ payrollType: "Domestic" })}>Domestic</button>
                <button className={`btn ${filters.payrollType === "Foreign" ? "btn-primary" : "btn-outline-secondary"}`} onClick={() => handleFiltersChange({ payrollType: "Foreign" })}>Foreign</button>
              </div>
            </div>
            <div className="col-md-6 col-lg-3">
              <label className="form-label fw-bold">Payment Currency</label>
              <select className="form-select" value={filters.paymentCurrency} onChange={e => handleFiltersChange({ paymentCurrency: e.target.value })}>
                <option value="">Select Currency</option>
                <option value="INR">INR</option>
                <option value="USD">USD</option>
                <option value="EUR">EUR</option>
              </select>
            </div>
          </div>

          {/* Add Employee Form */}
          <div className="my-4 p-4 border rounded-3 bg-light">
            <h5 className="mb-3">Add Employee Manually</h5>
            {error && <div className="alert alert-danger p-2">{error}</div>}
            <form onSubmit={handleAddEmployee} className="row g-3 align-items-end">
              <div className="col-sm-6 col-md-2"><label className="form-label">Employee ID</label><input type="text" name="id" value={newEmployee.id} onChange={handleInputChange} className="form-control" /></div>
              <div className="col-sm-6 col-md-3"><label className="form-label">Employee Name</label><input type="text" name="name" value={newEmployee.name} onChange={handleInputChange} className="form-control" /></div>
              <div className="col-sm-6 col-md-2"><label className="form-label">Salary</label><input type="text" name="salary" value={newEmployee.salary} onChange={handleInputChange} className="form-control" /></div>
              <div className="col-sm-6 col-md-2"><label className="form-label">Currency</label><select name="currency" value={newEmployee.currency} onChange={handleInputChange} className="form-select"><option>INR</option><option>USD</option><option>EUR</option></select></div>
              <div className="col-md-3"><label className="form-label">Account Number</label><input type="text" name="accountNumber" value={newEmployee.accountNumber} onChange={handleInputChange} className="form-control" /></div>
              <div className="col-12 text-end"><button type="submit" className="btn btn-primary">Add Employee</button></div>
            </form>
          </div>

          {/* Payroll Table */}
          <div className="mt-4">
            <div className="table-responsive">
              <table className="table table-bordered table-hover align-middle">
                <thead className="table-light"><tr><th>Employee ID</th><th>Employee Name</th><th>Salary Amount</th><th>Currency</th><th>Account Number</th><th className="text-center">Actions</th></tr></thead>
                <tbody>
                  {employees.length > 0 ? employees.map((e, idx) => (<tr key={idx}><td>{e.id}</td><td>{e.name}</td><td>{parseFloat(e.salary || 0).toFixed(2)}</td><td>{e.currency}</td><td>{e.accountNumber}</td><td className="text-center"><button className="btn btn-danger btn-sm" onClick={() => handleRemoveEmployee(idx)}>Delete</button></td></tr>)) : (<tr><td colSpan="6" className="text-center text-muted py-4">No employee data. Please add employees.</td></tr>)}
                </tbody>
              </table>
            </div>
            <div className="d-flex justify-content-end gap-4 fw-bold mt-3 p-3 bg-light rounded-3">
              <span>Total Employees: {employees.length}</span>
              <span>Total Debit Amount: {totalAmount.toFixed(2)}</span>
            </div>
          </div>

          {/* Actions */}
          <div className="mt-4 d-flex flex-wrap justify-content-center gap-3 border-top pt-4">
            <button className="btn btn-success" onClick={() => alert('Submitted for Approval')}>Submit for Approval</button>
            <button className="btn btn-outline-primary" onClick={() => alert('Saved as Draft')}>Save Draft</button>
            <button className="btn btn-danger" onClick={() => setEmployees([])}>Clear All</button>
            <button className="btn btn-secondary" onClick={() => alert('Cancelled')}>Cancel</button>
          </div>
        </div>
      </div>
    </>
  );
}

export default PayrollManual;
