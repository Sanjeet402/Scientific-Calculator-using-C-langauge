    <script>
        (function () {
            const form = document.getElementById('formTestimonials');
            const inputName = document.getElementById('inputName');
            const inputMsg = document.getElementById('inputMsg');
            const list = document.getElementById('listTestimonials');
            const hiddenEditId = document.getElementById('hiddenEditId');
            const btnSubmit = document.getElementById('btnSubmit');

            // --- Storage helpers ---
            function getData() {
                return JSON.parse(localStorage.getItem("testimonials") || "[]");
            }
            function setData(arr) {
                localStorage.setItem("testimonials", JSON.stringify(arr));
            }

            // utility to generate IDs
            function uid() { return 'id-' + Math.random().toString(36).slice(2, 9); }

            // Render testimonial HTML
            function buildTestimonial(id, name, msg) {
                const wrap = document.createElement('div');
                wrap.className = 'card-testimonial';
                wrap.dataset.id = id;

                const pill = document.createElement('div');
                pill.className = 'pill-edit';
                pill.innerHTML = 'Edit | <small>Delete</small>';
                pill.addEventListener('click', onPillClick);

                const nm = document.createElement('div');
                nm.className = 'field-name';
                nm.textContent = name;

                const mg = document.createElement('div');
                mg.className = 'field-msg';
                mg.textContent = msg;

                wrap.appendChild(pill);
                wrap.appendChild(nm);
                wrap.appendChild(mg);
                return wrap;
            }

            // Render all testimonials
            function renderAll() {
                list.innerHTML = "";
                const arr = getData();
                arr.forEach(t => {
                    const elem = buildTestimonial(t.id, t.name, t.msg);
                    list.appendChild(elem);
                });
            }

            // Handle form submit
            form.addEventListener('submit', function (e) {
                e.preventDefault();
                const name = inputName.value.trim();
                const msg = inputMsg.value.trim();
                if (!name || !msg) {
                    window.alert('⚠ Please enter both name and message.');
                    return;
                }

                const arr = getData();
                const editingId = hiddenEditId.value;

                if (editingId) {
                    const index = arr.findIndex(t => t.id === editingId);
                    if (index > -1) {
                        arr[index].name = name;
                        arr[index].msg = msg;
                    }
                    setData(arr);
                    hiddenEditId.value = "";
                    btnSubmit.textContent = "Submit";
                } else {
                    const id = uid();
                    arr.push({ id, name, msg });
                    setData(arr);
                }

                renderAll();
                form.reset();
            });

            // handle edit/delete pill click
            function onPillClick(e) {
                const parent = e.currentTarget.closest('.card-testimonial');
                const id = parent.dataset.id;

                const choice = confirm('Choose OK to Edit, Cancel to Delete.');
                if (choice) {
                    const arr = getData();
                    const item = arr.find(t => t.id === id);
                    if (item) {
                        inputName.value = item.name;
                        inputMsg.value = item.msg;
                        hiddenEditId.value = id;
                        btnSubmit.textContent = 'Update';
                        window.scrollTo({ top: 0, behavior: 'smooth' });
                    }
                } else {
                    if (confirm('Delete this testimonial?')) {
                        let arr = getData().filter(t => t.id !== id);
                        setData(arr);
                        renderAll();
                    }
                }
            }

            // --- On page load ---
            renderAll();
        })();
    </script>
